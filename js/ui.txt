<style>
  body {
    display: flex;
    justify-content: center;
    align-items: center;
    min-height: 100vh;
    background-color: #f0f2f5;
    margin: 0;
    font-family: sans-serif;
  }
  .container {
    position: relative;
    text-align: center;
  }
  canvas {
    position: absolute;
    top: 50%;
    left: 50%;
    transform: translate(-50%, -50%);
    pointer-events: none; /* Allows clicks to pass through to the button */
    z-index: 10;
  }
</style>

<div class="container">
  <canvas id="confettiCanvas"></canvas>
  <button id="upgradeButton" class="px-8 py-4 bg-yellow-400 text-purple-800 font-bold rounded-full shadow-lg hover:bg-yellow-500 transition duration-300 relative z-20">
    <span class="mr-2">⭐</span> 升级 (50)
  </button>
</div>

<script>
  const upgradeButton = document.getElementById('upgradeButton');
  const confettiCanvas = document.getElementById('confettiCanvas');
  const ctx = confettiCanvas.getContext('2d');

  let animationFrameId;
  let particles = [];
  const colors = ['#f44336', '#e91e63', '#9c27b0', '#673ab7', '#3f51b5', '#2196f3', '#03a9f4', '#00bcd4', '#009688', '#4caf50', '#8bc34a', '#cddc39', '#ffeb3b', '#ffc107', '#ff9800', '#ff5722'];

  function resizeCanvas() {
    confettiCanvas.width = window.innerWidth;
    confettiCanvas.height = window.innerHeight;
  }

  window.addEventListener('resize', resizeCanvas);
  resizeCanvas();

  class ConfettiParticle {
    constructor(x, y, color) {
      this.x = x;
      this.y = y;
      this.size = Math.random() * 8 + 3;
      this.color = color;
      this.velocity = {
        x: (Math.random() - 0.5) * 8,
        y: Math.random() * -15 - 5
      };
      this.rotation = Math.random() * 360;
      this.rotationSpeed = Math.random() * 10 - 5;
      this.alpha = 1;
      this.gravity = 0.5;
      this.friction = 0.99;
      this.decay = 0.02;
    }

    draw() {
      ctx.save();
      ctx.translate(this.x, this.y);
      ctx.rotate(this.rotation * Math.PI / 180);
      ctx.fillStyle = this.color;
      ctx.globalAlpha = this.alpha;
      ctx.fillRect(-this.size / 2, -this.size / 2, this.size, this.size);
      ctx.restore();
    }

    update() {
      this.velocity.x *= this.friction;
      this.velocity.y += this.gravity;
      this.x += this.velocity.x;
      this.y += this.velocity.y;
      this.rotation += this.rotationSpeed;
      this.alpha -= this.decay;
    }
  }

  function createConfetti(x, y) {
    for (let i = 0; i < 50; i++) {
      const color = colors[Math.floor(Math.random() * colors.length)];
      particles.push(new ConfettiParticle(x, y, color));
    }
  }

  function animateConfetti() {
    ctx.clearRect(0, 0, confettiCanvas.width, confettiCanvas.height);
    for (let i = particles.length - 1; i >= 0; i--) {
      particles[i].update();
      particles[i].draw();

      if (particles[i].alpha <= 0 || particles[i].y > confettiCanvas.height) {
        particles.splice(i, 1);
      }
    }

    if (particles.length > 0) {
      animationFrameId = requestAnimationFrame(animateConfetti);
    } else {
      cancelAnimationFrame(animationFrameId);
    }
  }

  upgradeButton.addEventListener('click', () => {
    upgradeButton.innerHTML = '<span class="mr-2">⭐</span> 升级成功';
    upgradeButton.classList.remove('bg-yellow-400', 'hover:bg-yellow-500');
    upgradeButton.classList.add('bg-green-500', 'hover:bg-green-600', 'text-white'); // Optional: change button color

    // Get button position to emit confetti from it
    const rect = upgradeButton.getBoundingClientRect();
    const centerX = rect.left + rect.width / 2;
    const centerY = rect.top + rect.height / 2;

    createConfetti(centerX, centerY);
    if (!animationFrameId) { // Only start animation if not already running
      animateConfetti();
    }
  });
</script>
