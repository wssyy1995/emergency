    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Canvas Game Buttons</title>
    <style>
        body {
            margin: 0;
            display: flex;
            justify-content: center;
            align-items: center;
            min-height: 100vh;
            background-color: #F0F5FA;
            font-family: sans-serif;
        }
        canvas {
            background-color: #ffffff;
            border-radius: 24px;
            box-shadow: 0 10px 30px rgba(0,0,0,0.05);
            cursor: default; /* 我们将在 JS 中动态控制鼠标样式 */
        }
    </style>

<body>

<canvas id="gameCanvas" width="500" height="300"></canvas>

<script>
    const canvas = document.getElementById('gameCanvas');
    const ctx = canvas.getContext('2d');

    // 按钮类定义
    class CanvasButton {
        constructor(x, y, width, height, text, style) {
            this.x = x;
            this.y = y;
            this.w = width;
            this.h = height;
            this.text = text;
            
            // 样式配置
            this.style = style;
            
            // 状态
            this.isHovered = false;
            this.isPressed = false;
        }

        // 碰撞检测（点是否在矩形内）
        isHit(mouseX, mouseY) {
            return mouseX >= this.x && mouseX <= this.x + this.w &&
                   mouseY >= this.y && mouseY <= this.y + this.h;
        }

        draw(ctx) {
            const radius = 16; // 圆角半径
            // 根据是否按下，计算顶层的 Y 轴偏移量 (模拟按下效果)
            const pressOffsetY = this.isPressed ? this.style.thickness : 0;
            
            // 1. 绘制底层阴影/厚度 (固定的)
            ctx.fillStyle = this.style.bottomColor;
            ctx.beginPath();
            ctx.roundRect(this.x, this.y + this.style.thickness, this.w, this.h, radius);
            ctx.fill();

            // 2. 绘制顶层 (会随按下状态移动)
            // 如果处于 hover 状态且未按下，颜色稍微变亮一点（这里用一个简单的蒙版模拟）
            ctx.fillStyle = this.style.topColor;
            ctx.beginPath();
            ctx.roundRect(this.x, this.y + pressOffsetY, this.w, this.h, radius);
            ctx.fill();

            // Hover 时的白色高光叠加
            if (this.isHovered && !this.isPressed) {
                ctx.fillStyle = 'rgba(255, 255, 255, 0.15)';
                ctx.fill();
            }

            // 3. 绘制文字
            ctx.fillStyle = this.style.textColor;
            ctx.font = 'bold 16px sans-serif';
            ctx.textAlign = 'center';
            ctx.textBaseline = 'middle';
            // 文字位置跟着顶层一起移动
            ctx.fillText(this.text, this.x + this.w / 2, this.y + this.h / 2 + pressOffsetY);
            
            // 如果是描边风格的按钮（清空按钮）
            if (this.style.isStroke) {
                ctx.strokeStyle = this.style.topColor;
                ctx.lineWidth = 2;
                ctx.stroke();
            }
        }
    }

    // --- 实例化按钮 ---
    // 发送按钮：你喜欢的天蓝色 Q版立体风格
    const btnSend = new CanvasButton(280, 120, 140, 48, "发 送", {
        topColor: '#38BDF8',       // 顶层主色
        bottomColor: '#0284C7',    // 底层厚度色
        textColor: '#FFFFFF',      // 文字颜色
        thickness: 6,              // 3D厚度 (按下时移动的距离)
        isStroke: false
    });

    // 清空按钮：灰色低存在感，防误触风格
    const btnClear = new CanvasButton(80, 120, 120, 48, "清 空", {
        topColor: '#E2E8F0',       // 顶层主色 (浅灰)
        bottomColor: '#CBD5E1',    // 底层厚度色 (深灰)
        textColor: '#64748B',      // 文字颜色
        thickness: 6,
        isStroke: false
    });

    const buttons = [btnClear, btnSend];

    // --- 渲染循环 ---
    function render() {
        // 清空画布
        ctx.clearRect(0, 0, canvas.width, canvas.height);
        
        // 画个简单的背景装饰
        ctx.fillStyle = '#F8FAFC';
        ctx.beginPath();
        ctx.roundRect(40, 40, 420, 220, 20);
        ctx.fill();
        ctx.strokeStyle = '#E2E8F0';
        ctx.lineWidth = 2;
        ctx.stroke();

        ctx.fillStyle = '#94A3B8';
        ctx.font = '14px sans-serif';
        ctx.textAlign = 'center';
        ctx.fillText('Canvas 交互演示区', canvas.width / 2, 80);

        // 绘制所有按钮
        buttons.forEach(btn => btn.draw(ctx));
    }

    // --- 事件监听 (核心逻辑) ---
    // 获取鼠标在 Canvas 中的相对坐标
    function getMousePos(evt) {
        const rect = canvas.getBoundingClientRect();
        return {
            x: evt.clientX - rect.left,
            y: evt.clientY - rect.top
        };
    }

    // 鼠标移动事件：处理 Hover 状态和鼠标指针样式
    canvas.addEventListener('mousemove', (e) => {
        const pos = getMousePos(e);
        let cursorStyle = 'default';

        buttons.forEach(btn => {
            if (btn.isHit(pos.x, pos.y)) {
                btn.isHovered = true;
                cursorStyle = 'pointer'; // 变成小手
            } else {
                btn.isHovered = false;
                btn.isPressed = false; // 移出按钮区域时重置按压状态
            }
        });

        canvas.style.cursor = cursorStyle;
        render(); // 状态改变，重新绘制
    });

    // 鼠标按下事件：处理 Pressed 状态
    canvas.addEventListener('mousedown', (e) => {
        const pos = getMousePos(e);
        buttons.forEach(btn => {
            if (btn.isHit(pos.x, pos.y)) {
                btn.isPressed = true;
            }
        });
        render();
    });

    // 鼠标抬起事件：触发点击逻辑并恢复状态
    canvas.addEventListener('mouseup', (e) => {
        const pos = getMousePos(e);
        buttons.forEach(btn => {
            if (btn.isPressed && btn.isHit(pos.x, pos.y)) {
                // 这里触发按钮的点击事件
                if(btn.text === '发 送') {
                    alert('你点击了【发送】按钮！触发 Canvas 发送逻辑。');
                } else {
                    alert('你点击了【清空】按钮！');
                }
            }
            btn.isPressed = false;
        });
        render();
    });

    // 鼠标离开画布区域，重置所有状态
    canvas.addEventListener('mouseleave', () => {
        buttons.forEach(btn => {
            btn.isHovered = false;
            btn.isPressed = false;
        });
        canvas.style.cursor = 'default';
        render();
    });

    // 初始渲染
    render();

</script>
</body>
